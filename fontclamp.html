/* すべての要素に対してデフォルト設定 (rem基準) */
*,
*::before,
*::after {
  --root-size: 16; /* 1rem = 16px */
  --base-size: 0.75; /* 基準フォントサイズ(rem) [12px相当] */
  --growth-rate: 2; /* 変化率 */
  --min-width: 20; /* 最小画面幅(rem) [320px相当] */
  --max-width: 45; /* 最大画面幅(rem) [720px相当] */

  /* 計算部分 */
  --fz: 1; /* デフォルト倍率 */
  --min-size: calc(var(--base-size) * var(--fz, 1)); /* 最小サイズ(rem) */
  --max-size: calc(var(--min-size) * var(--growth-rate)); /* 最大サイズ(rem) */
  
  /* 傾き(vw)の計算: (max_rem - min_rem) / (max_width_rem - min_width_rem) * 100 */
  --slope: calc(
    (var(--max-size) - var(--min-size)) / (var(--max-width) - var(--min-width)) * 100vw
  );
  
  /* 切片(rem)の計算 */
  --intercept: calc(
    var(--min-size) - 
    ((var(--max-size) - var(--min-size)) / (var(--max-width) - var(--min-width)) * var(--min-width))
  );

  /* フォントサイズの計算 */
  --font-size: clamp(
    calc(var(--min-size) * 1rem),
    calc(var(--intercept) * 1rem + var(--slope)),
    calc(var(--max-size) * 1rem)
  );
}

@media (min-width: 45rem) { /* 720px相当 */
  *,
  *::before,
  *::after {
    --base-size: 1; /* 基準フォントサイズ(rem) [16px相当] */
    --growth-rate: 1.5; /* 変化率 */
    --min-width: 45; /* 最小画面幅(rem) [720px相当] */
    --max-width: 80; /* 最大画面幅(rem) [1280px相当] */
  }
}
