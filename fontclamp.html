/* すべての要素に対してデフォルト設定 (rem基準) */
*,
*::before,
*::after {
  --min-size-rem: 1;    /* 最小文字サイズ(rem) */
  --max-size-rem: 2;    /* 最大文字サイズ(rem) */
  --min-width-rem: 20;  /* 最小画面幅(rem) [320px相当] */
  --max-width-rem: 45;  /* 最大画面幅(rem) [720px相当] */

  /* 傾き(vw)の計算 */
  --slope: calc(
    (var(--max-size-rem) - var(--min-size-rem)) / (var(--max-width-rem) - var(--min-width-rem)) * 100vw
  );
  
  /* 切片(rem)の計算 */
  --intercept: calc(
    var(--min-size-rem) - 
    ((var(--max-size-rem) - var(--min-size-rem)) / (var(--max-width-rem) - var(--min-width-rem)) * var(--min-width-rem))
  );

  /* フォントサイズの計算 */
  --font-size: clamp(
    calc(var(--min-size-rem) * 1rem),
    calc(var(--intercept) * 1rem + var(--slope)),
    calc(var(--max-size-rem) * 1rem)
  );
}

@media (min-width: 45rem) { /* 720px相当 */
  *,
  *::before,
  *::after {
    --min-size-rem: 1.25; /* 最小文字サイズ(rem) */
    --max-size-rem: 2.5;  /* 最大文字サイズ(rem) */
    --min-width-rem: 45;  /* 最小画面幅(rem) [720px相当] */
    --max-width-rem: 80;  /* 最大画面幅(rem) [1280px相当] */
  }
}
