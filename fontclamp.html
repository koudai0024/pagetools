/* すべての要素に対してデフォルト設定 (文字rem / 画面px) */
*,
*::before,
*::after {
  --min-size-rem: 1;    /* 最小文字サイズ(rem) */
  --max-size-rem: 2;    /* 最大文字サイズ(rem) */
  --min-width-px: 320;  /* 最小画面幅(px) */
  --max-width-px: 720;  /* 最大画面幅(px) */

  /* 傾き(vw)の計算: (max_rem - min_rem) * 16 / (max_width_px - min_width_px) * 100 */
  --slope: calc(
    (var(--max-size-rem) - var(--min-size-rem)) * 16 / (var(--max-width-px) - var(--min-width-px)) * 100vw
  );
  
  /* 切片(rem)の計算 */
  --intercept: calc(
    var(--min-size-rem) - 
    ((var(--max-size-rem) - var(--min-size-rem)) * 16 / (var(--max-width-px) - var(--min-width-px)) * var(--min-width-px) / 16)
  );

  /* フォントサイズの計算 */
  --font-size: clamp(
    calc(var(--min-size-rem) * 1rem),
    calc(var(--intercept) * 1rem + var(--slope)),
    calc(var(--max-size-rem) * 1rem)
  );
}

@media (min-width: 720px) {
  *,
  *::before,
  *::after {
    --min-size-rem: 1.25; /* 最小文字サイズ(rem) */
    --max-size-rem: 2.5;  /* 最大文字サイズ(rem) */
    --min-width-px: 720;  /* 最小画面幅(px) */
    --max-width-px: 1280; /* 最大画面幅(px) */
  }
}
