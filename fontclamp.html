/* すべての要素に対してデフォルト設定 */
*,
*::before,
*::after {
  --min-size-rem: 1;    /* 最小文字サイズ(rem) */
  --max-size-rem: 2;    /* 最大文字サイズ(rem) */
  --min-width-px: 320;  /* 最小画面幅(px) */
  --max-width-px: 720;  /* 最大画面幅(px) */

  /* 【重要】1remを固定しない計算
    傾き（slope）は、文字(rem)の差分を画面(px)の差分で割り、100vwを掛けます。
    （単位としては [rem/px * vw] になりますが、CSSの計算上は数値として処理されます）
  */
  --slope: calc(
    (var(--max-size-rem) - var(--min-size-rem)) / (var(--max-width-px) - var(--min-width-px)) * 100vw
  );
  
  /* 切片（intercept）の計算
    最小サイズ(rem) から「傾き × 最小画面幅(px)」を引きます。
    ここでもpxとremが混在するため、1remを掛け合わせて単位をremに統一します。
  */
  --intercept: calc(
    var(--min-size-rem) * 1rem - 
    ((var(--max-size-rem) - var(--min-size-rem)) / (var(--max-width-px) - var(--min-width-px)) * var(--min-width-px) * 1rem)
  );

  /* フォントサイズの適用 */
  --font-size: clamp(
    calc(var(--min-size-rem) * 1rem),
    calc(var(--intercept) + var(--slope)),
    calc(var(--max-size-rem) * 1rem)
  );
}
