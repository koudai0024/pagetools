/* すべての要素に対してデフォルト設定 */
*,
*::before,
*::after {
  --min-size: 1rem;   /* 最小文字サイズ(rem) */
  --max-size: 1.5rem; /* 最大文字サイズ(rem) */
  --min-width: 720;   /* 最小画面幅(px) */
  --max-width: 1280;  /* 最大画面幅(px) */

  /* 計算部分 (すべて純粋な数値・remとして計算) */
  --slope: calc(
    (var(--max-size) - var(--min-size)) / (var(--max-width) - var(--min-width)) * 100vw
  );
  --intercept: calc(
    var(--min-size) -
      (
        (var(--max-size) - var(--min-size)) /
          (var(--max-width) - var(--min-width)) * var(--min-width)
      )
  );

  /* フォントサイズの計算 */
  --font-size: clamp(
    var(--min-size),
    calc(var(--intercept) + var(--slope)),
    var(--max-size)
  );
}
